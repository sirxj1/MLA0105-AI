diagnose_patient(Patient, Condition) :-
    symptom(Patient, Symptom),
    condition(Symptom, Condition).

diagnose_interactive :-
    write('Enter patient name: '),
    read(Patient),
    write('Enter symptom: '),
    read(Symptom),
    condition(Symptom, Condition),
    write('Diagnosis for '), write(Patient), write(': '), write(Condition),
    nl.
